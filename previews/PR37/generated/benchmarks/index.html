# load packages
using BenchmarkTools
using WaterPhysics
FT = Float32;
# define the variables
rand_r = (rand(FT) + 20) * FT(1e-6);
rand_T = rand(FT) + 298;
rand_α = rand(FT) * 50;
rand_Ψ = rand(FT) - 3;

gas_air = TraceGasAir();
gas_CO₂ = TraceGasCO₂();

println("\nBenchmarking capillary_pressure functions...");
@btime capillary_pressure($rand_r, $rand_T);
@btime capillary_pressure($rand_r, $rand_T, $rand_α);
